<script>
function StartScripts()
{
	GeoLocate()
}
function GeoLocate()
{
	if (navigator.geolocation) { 
		navigator.geolocation.getCurrentPosition(showLocation)
    }
}
function showLocation(position) {
	var latitude = position.coords.latitude;
	var longitude = position.coords.longitude;
	document.getElementById("sLoc").value = latitude + " " + longitude
}

function Validate()
{
	var sn = document.forms["myForm"]["sName"].value
	var sl = document.forms["myForm"]["sLoc"].value
	var se = document.forms["myForm"]["sEmail"].value
	var sm = document.forms["myForm"]["sMsg"].value
	if (sn == "" || sl == "" || se == "" || sm == "")
	{
		alert("Please fill in the empty text boxes")
		return false
	}
}
</script>
